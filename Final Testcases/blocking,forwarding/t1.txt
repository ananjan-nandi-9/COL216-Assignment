main:
	addi $t0, $zero, 0
	addi $t1, $zero, 1
	addi $s0, $zero, 10
	sw $t0, 1000($zero)
	sw $t1, 2000($zero)

fibonacci:
	lw $t0, 1000($zero)
	lw $t1, 2000($zero)
	add $t1, $t0, $t1	
	sub $t0, $t1, $t0
	sw $t0, 1000($zero)
	sw $t1, 2000($zero)
	addi $s0, $s0, -1
	bne $s0, $zero, fibonacci
exit:
